Prompt 'Tape'
wait 1
Prompt ''

If
	$Stage = 3
Then
	ZoomSpot Menu.Question 'Prepare Tape?'
	ZoomSpot Menu.Choices 'Yes','No'
	ZoomSpot Menu.on ZoomSpot
	WaitFor
	ZoomSpot Menu.Done
	Wait .1

	If
		ZoomSpot menu.result 'Yes'
	Then
		Prompt 'Tape is prepped.'
		wait 1
		Prompt ''
	Endif
Endif

If
	$Stage = 4
Then
	ZoomSpot Menu.Question 'Use tape to anchor catheter?'
	ZoomSpot Menu.Choices 'Yes','No'
	ZoomSpot Menu.on ZoomSpot
	WaitFor
	ZoomSpot Menu.Done
	Wait .1

	If
		ZoomSpot menu.result 'Yes'
	Then
		Prompt 'Catheter is anchored.'
		wait 1
		Prompt ''
	Endif
Endif
