# 'OpenCatheter'
Prompt 'OpenCatheter'
wait .51
Prompt ''

If
	Do 'AnswerIs' 'Open catheter?' 'No'
Then
	Return
Endif



#setup variables for function calls
$Item = Catheter
$ItemObj = $CatheterBalloon
$ItemPos = CatheterPos
$CatheterBalloon = ItemObj + 'Instance'  #DropFromPackage adds Instance to name



#Grab item, and tear package open.
Do 'UCGActions/GrabAndOpenPackage' $Item

#pour package
Do 'UCGActions/PourPackage' $Item $ItemPos 0.15

#drop object from package
Do 'UCGActions/DropFromPackage' $Item $ItemObj $ItemPos

Wait 2

#Stop pour
Do 'UCGActions/StopPackagePour' $Item 

#Throw away package
Do 'UCGActions/ThrowAwayPackage' $Item 

#back player up
$Player localMoveZ -0.1 0.5
