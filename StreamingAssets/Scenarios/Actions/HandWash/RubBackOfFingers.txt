    $param[0] > 0
Then
    $whichHand = param[1]
Else
    $whichHand = 'right'
Endif

Wait 2

$othermotionType = 'moveforward'

If
    $whichHand == 'right'
Then
    $MainHandTarget = NurseLeftHandDorsum
Else
    $MainHandTarget = NurseRightHandDorsum
Endif

$MainHandAlignment = 'default' #'rotateleft'
$StartAtTop = true
Do 'Actions/HandWash/RubHands.txt' $whichHand $MainHandTarget 'pond' $MainHandAlignment $mainmotionType 50.09 $HandPalm 'MiddleHandSpot' 'grip' 'default' $othermotionType 0.001 $HandPalm  5 .5 60 0 0
