#PatientIdle - main patient Loop
# $Patient - patient object
# $WristTag
# $IDLabel

Debug 'PatientIdle:Entering'
Player align $PlayerLookPat
#Player lookAt $PlayerLookPat
Debug 'PatientIdle: align PlayerLookPos'
#Wait 5
#Player lookat $PlayerPatPos
JoystickCanvas on
PatientCanvas off

$Patient reset
$WristTag reset
$IDLabel reset
$GXMForm reset
$InfusionSet reset

$Patient outline.on
$WristTag outline.on
$IDLabel outline.on
$GXMForm outline.on
$InfusionSet outline.on

Label 'Loop'

	Debug 'MainLoop:Loop'
	$onOff = false
#	Debug $PC_ExitGame

	If
		$Nurse pressed
	Then
		$Nurse resetPressed
#		Do '05/AllClicks.txt'
		Do '05/WaitForNurseCall.txt'
		$onOff = true
		Do '05/AllClicks.txt'
	Endif

	If
		$Patient pressed
	Then
		$Patient resetPressed
		Do '05/AllClicks.txt'
		Do '05/PatientTasks.txt'
		$Patient reset
		$onOff = true
		Do '05/AllClicks.txt'
	Endif



	If
		$WristTag pressed
	Then
		$WristTag resetPressed
		Do '05/AllClicks.txt'
		Do '05/CheckWristTag.txt'
		$WristTag resetPressed
		Player align $PlayerLookPat .2
		$onOff = true
		Do '05/AllClicks.txt'
	Endif
	Wait .01

	If
		$IDLabel pressed
	Then
		$IDLabel resetPressed
		Do '05/AllClicks.txt'
		Do '05/CheckLabel.txt'
		$onOff = true
		Do '05/AllClicks.txt'
		Player align $PlayerLookCow
	Endif

	Wait .01

	If
		$GXMForm pressed
	Then
		$GXMForm resetPressed
		Do '05/AllClicks.txt'
		Do '05/CheckGXMForm.txt'
		$onOff = true
		Do '05/AllClicks.txt'
		Player align $PlayerLookCow
	Endif

	If
		$InfusionSet pressed
	Then
		$InfusionSet resetPressed
		Do '05/AllClicks.txt'
		$BloodCheck2 = true
		Do '05/CheckBag.txt'
		$onOff = true
		Do '05/AllClicks.txt'
	Endif
	If
		$ChecklistButton pressed
	Then
		$ChecklistButton resetPressed
		Do '05/AllClicks.txt'
		Do 'ShowList.txt'
	Endif

	If
		$Scene_Exit pressed
	Then 
		$Scene_Exit resetPressed
		Do 'EndCheck.txt'
	Endif
	If
		$SumButton pressed
	Then 
		$SceneStart = '05/MainLoop.txt'
		$SumButton resetPressed
		Do 'SummaryCheck.txt'
	Endif
	Wait .01
Goto 'Loop'
